int sw = 2;
int led = 12;

void setup()
{
  pinMode(sw, INPUT);
  pinMode(led, OUTPUT);
}

void loop()
{
  if (digitalRead(sw) == HIGH){
    digitalWrite(led, HIGH);
  }
  else {
    digitalWrite(led, LOW);
  }
}
