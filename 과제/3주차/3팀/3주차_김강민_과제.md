```cpp
int led = 13;
int sw = 2;
void setup()
{
  pinMode(led, OUTPUT);
  pinMode(sw,INPUT); #인풋, 아웃풋 포트 설정
}

void loop()
{
  if(digitalRead(sw) == HIGH){ #신호가 HIGH일 때
	digitalWrite(led,HIGH); #LED HIGH로 활성화
  }
  else{ #그 외의 경우(LOW일 때)
  	digitalWrite(led,LOW); #LED LOW로 비활성화
  }

}
```
